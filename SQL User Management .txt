1. create user :
create user d2 identified by sit;
#user d2 is created with password sit
#user d2 needs permissions for database operations

2. grant permissions :
grant connect, resource to d2;
#permissions granted to user d2

3. change password :
alter user d2 identified by newpassword;
#password can contain only 0-9, a-z, A-Z
#special characters are not allowed

4. login :
enter username as: d2
enter password as: newpassword
#user d2 is now connected to the database

5. commit or exit :
commit;
#saves the changes

OR

exit;
#saves and disconnects from the database
